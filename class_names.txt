1. GraphObj -> DiGraph
// класс объекта "направленный граф (Direction Graph)"
2. Preprocessor -> SaymonDataPreprocessor
// класс предобработки данных из платформы SAYMON
3. Handler -> OracleHandler
// класс для манипулирования БД Oracle через интерфейс Python
4. Schema -> TablesSchema
// класс хранящий схемы данных различных таблиц
5. Forecaster -> ForecastModel
// класс прогнозной модели машинного обучения

2. fit/train -> fit

// объект для логирования информации о работе программы
